#include <stdio.h>
#include <stdlib.h>

void dijkstras(int n,int s,int dist[10],int cost[10][10]){
    int i,v,count=1,min,visited[10];
    for(i=1;i<=n;i++){
        visited[i]=0;
        dist[i]=cost[s][i];
    }
    visited[s]=1;
    dist[i]=0;
    while(count<n){
        min=999;
        for(i=1;i<=n;i++){
            if(dist[i]<min && visited[i]==0){
                min=dist[i];
                v=i;
            }
        }
        visited[v]=1;
        count++;
        for(i=1;i<=n;i++)
        if(dist[i]>dist[v]+cost[v][i])
        dist[i]=dist[v]+cost[v][i];
    }
}
